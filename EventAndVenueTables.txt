	Event_ID int NOT NULL,
	Venue_ID int NOT NULL,
	Name varchar(40) NOT NULL,
	EDate DATE,
	PRIMARY KEY (Event_ID)
);

	Venue_ID int NOT NULL,
	Name varchar(40) NOT NULL,
